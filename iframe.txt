    iframe { width: 100%; height: 100%; border: none; }
  </style>

<body>
  <iframe id="mapFrame"></iframe>

  <script>
    function getParam(key, fallback = '') {
      const params = new URLSearchParams(window.location.search);
      return params.get(key) || fallback;
    }

    const lat = getParam('lat', '32.613899');
    const lon = getParam('lon', '-83.687934');
    const z = getParam('z', '17');
    const t = getParam('t', 'h');

    const embedUrl = `https://maps.google.com/maps?q=${lat},${lon}&z=${z}&t=${t}&output=svembed`;
    document.getElementById('mapFrame').src = embedUrl;
  </script>
